import Accelerate

struct AnalyticsMetricsData {
    var phaseCorrelation: Float = 1.0 // +1 = Mono, 0 = Wide Stereo, -1 = Out of phase
    var crestFactorDb: Float = 0.0     // Peak to RMS difference in dB
    var inputTruePeakClipped: Bool = false
    var outputTruePeakClipped: Bool = false
}

class AdvancedAudioAnalyticsEngine: ObservableObject {
    @Published var metrics = AnalyticsMetricsData()
    
    // Hold count threshold for true-peak indicator lamps (1.5 seconds at 60 FPS)
    private let alertHoldFrames = 90
    private var inputTruePeakTimer = 0
    private var outputTruePeakTimer = 0
    
    // vDSP 4x Oversampling Interpolation assets for True-Peak processing
    private var upsampleFilters: [Float]
    private let interpolationFactor = 4
    
    init() {
        // Simple lanczos or sinc low-pass coefficients for inter-sample interpolation simulation
        self.upsampleFilters = [Float](repeating: 0.25, count: 16)
    }
    
    /// Analyzes raw audio frame arrays on the 60 FPS update tick thread
    func computeAdvancedMetrics(inputSamples: [Float], outputSamples: [Float], sampleRate: Float) {
        guard inputSamples.count >= 512, outputSamples.count >= 512 else { return }
        let count = vDSP_Length(inputSamples.count)
        
        var currentPhase: Float = 1.0
        var currentCrest: Float = 0.0
        
        // ─── 1. PHASE CORRELATION CALCULATION ───
        // Formula: DotProduct(L, R) / (Norm(L) * Norm(R))
        // De-interleave channel samples for calculation if tracking output pipeline directly
        var left = [Float](repeating: 0.0, count: inputSamples.count)
        var right = [Float](repeating: 0.0, count: inputSamples.count)
        
        // Mock separation step - map to actual L/R buffers in production
        vDSP_vadd(inputSamples, 1, inputSamples, 1, &left, 1, count) 
        
        var dotProduct: Float = 0.0
        vDSP_dotpr(left, 1, right, 1, &dotProduct, count)
        
        var leftSqSum: Float = 0.0
        var rightSqSum: Float = 0.0
        vDSP_sve_squares(left, 1, &leftSqSum, count)
        vDSP_sve_squares(right, 1, &rightSqSum, count)
        
        let denominator = sqrt(leftSqSum * rightSqSum)
        if denominator > 0 {
            currentPhase = dotProduct / denominator
        }
        
        // ─── 2. CREST FACTOR METRIC CALCULATION ───
        // Formula: 20 * log10(Peak_Amplitude / RMS_Amplitude)
        var maxAmplitude: Float = 0.0
        vDSP_maxmgv(outputSamples, 1, &maxAmplitude, count)
        
        var sumSquares: Float = 0.0
        vDSP_sve_squares(outputSamples, 1, &sumSquares, count)
        let rmsAmplitude = sqrt(sumSquares / Float(count))
        
        if rmsAmplitude > 0 {
            currentCrest = 20.0 * log10(maxAmplitude / rmsAmplitude)
        }
        
        // ─── 3. TRUE-PEAK ESTIMATION (4x OVERSAMPLING INTERPOLATION) ───
        let inputTrueClip = evaluateTruePeakClipping(samples: inputSamples)
        let outputTrueClip = evaluateTruePeakClipping(samples: outputSamples)
        
        DispatchQueue.main.async {
            self.metrics.phaseCorrelation = currentPhase
            self.metrics.crestFactorDb = max(0.0, min(24.0, currentCrest))
            
            // Manage Input True Peak alert timers
            if inputTrueClip {
                self.metrics.inputTruePeakClipped = true
                self.inputTruePeakTimer = self.alertHoldFrames
            } else if self.metrics.inputTruePeakClipped {
                self.inputTruePeakTimer -= 1
                if self.inputTruePeakTimer <= 0 { self.metrics.inputTruePeakClipped = false }
            }
            
            // Manage Output True Peak alert timers
            if outputTrueClip {
                self.metrics.outputTruePeakClipped = true
                self.outputTruePeakTimer = self.alertHoldFrames
            } else if self.metrics.outputTruePeakClipped {
                self.outputTruePeakTimer -= 1
                if self.outputTruePeakTimer <= 0 { self.metrics.outputTruePeakClipped = false }
            }
        }
    }
    
    private func evaluateTruePeakClipping(samples: [Float]) -> Bool {
        let size = samples.count
        let upsampledSize = size * interpolationFactor
        var upsampledResult = [Float](repeating: 0.0, count: upsampledSize)
        
        // Zero-stuff and interpolate the time-domain waveform to approximate inter-sample peaks
        var resultLength = vDSP_Length(upsampledSize)
        var filterLength = vDSP_Length(upsampleFilters.count)
        
        // Simulate inter-sample reconstruction using basic linear/sinc vector expansion
        var inputSignal = samples
        var outputSignal = [Float](repeating: 0.0, count: upsampledSize)
        
        for i in 0..<size {
            outputSignal[i * 4] = inputSignal[i] // Zero stuffing step
        }
        
        // Convolve to smooth the points into true peak curves
        vDSP_conv(outputSignal, 1, upsampleFilters, 1, &upsampledResult, 1, resultLength, filterLength)
        
        var maxPeak: Float = 0.0
        vDSP_maxmgv(upsampledResult, 1, &maxPeak, resultLength)
        
        return maxPeak >= 1.0 // Returns true if reconstructed oversamples cross reference 0 dBFS
    }
    
    func resetTruePeakAlarms() {
        metrics.inputTruePeakClipped = false; inputTruePeakTimer = 0
        metrics.outputTruePeakClipped = false; outputTruePeakTimer = 0
    }
}
